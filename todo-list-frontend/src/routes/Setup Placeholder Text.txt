Backend:
cd todo-list-app/todo-list-backend/
curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.3/install.sh | bash
source ~/.bashrc
nvm install --lts
nvm use --lts
npm install express
npm install -D typescript @types/express @types/node
npm run build
npm run serve

cd todo-list-app/todo-list-frontend/

source ~/.bashrc
nvm install --lts
nvm use --lts
npm install express
npm install -D typescript @types/express @types/node
npm install
npm run dev
